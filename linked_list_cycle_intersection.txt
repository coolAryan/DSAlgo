public class Solution {
    public ListNode detectCycle(ListNode head) {
          ListNode curr=head;
          ListNode slow =head;
          ListNode fast=head;
        
        while(slow!=null && fast!=null && fast.next!=null)
        {
            slow=slow.next;
            fast=fast.next.next;
              if(slow==fast)
              {
            
                  while(curr!=slow)
                  {
                     curr=curr.next;
                     slow=slow.next;
                  }
                  return slow;
              }
        }
    
          return null;
        
    }
}
